342 int eth_raw_init()
343 {
344 
345 
346  /* Vars */
347  int i;
348  struct ifreq ifr;
349  int ifindex = 0; /*Ethernet Interface index*/
350 
351  /* Open socket */
352  raw_socket = socket(PF_PACKET, SOCK_RAW, htons(ETH_TYPE));
353  if (raw_socket == -1)
354  {
355  perror("socket():");
356  exit(1);
357  }
358  ROS_INFO("Successfully opened socket: %i\n", raw_socket);
359 
360  /*Get ethernet interface index*/
361  strncpy(ifr.ifr_name, (const char*) interface, IFNAMSIZ);
362  if (ioctl(raw_socket, SIOCGIFINDEX, &ifr) == -1)
363  {
364  perror("SIOCGIFINDEX");
365  exit(1);
366  }
367  ifindex = ifr.ifr_ifindex;
368  ROS_INFO("Successfully got interface index: %i\n", ifindex);
369 
370  /*retrieve corresponding MAC*/
371  if (ioctl(raw_socket, SIOCGIFHWADDR, &ifr) == -1)
372  {
373  perror("SIOCGIFINDEX");
374  exit(1);
375  }
376 
377  if (mac_as_string.compare("") == 0)
378  {
379  for (i = 0; i < 6; i++)
380  {
381  src_mac[i] = ifr.ifr_hwaddr.sa_data[i];
382  }
383  ROS_INFO("Host MAC address: %02X:%02X:%02X:%02X:%02X:%02X\n",
384  src_mac[0], src_mac[1], src_mac[2], src_mac[3], src_mac[4], src_mac[5]);
385  }
386 
387  /*prepare sockaddr_ll*/
388  socket_address.sll_family = PF_INET;
389  socket_address.sll_protocol = htons(ETH_TYPE);
390  socket_address.sll_ifindex = ifindex;
391  //socket_address.sll_hatype = ARPHRD_ETHER;
392  socket_address.sll_pkttype = PACKET_BROADCAST;
393  socket_address.sll_halen = ETH_ALEN;
394  socket_address.sll_addr[0] = bcast_mac[0];
395  socket_address.sll_addr[1] = bcast_mac[1];
396  socket_address.sll_addr[2] = bcast_mac[2];
397  socket_address.sll_addr[3] = bcast_mac[3];
398  socket_address.sll_addr[4] = bcast_mac[4];
399  socket_address.sll_addr[5] = bcast_mac[5];
400  socket_address.sll_addr[6] = 0x00;
401  socket_address.sll_addr[7] = 0x00;
402 
403  return 0;
404 
405 }

424 bool getRoute(routing_entry& r, std::string hostname)
425 {
426  /* Description:
427  * Search a route to the destination host in the routing table.
428  * The parameter of the type routing_entry will be initialized, if a route can be found.
429  *
430  * Returns:
431  * (true) If route has been found
432  *
433  * (false) If route has not been found
434  */
435  boost::unique_lock<boost::mutex> lock(mtx_routing_table);
436 
437  for (std::list<routing_entry>::iterator it = routing_table_l.begin(); it != routing_table_l.end(); ++it)
438  {
439 
440  if ((*it).hostname_destination.compare(hostname) == 0)
441  {
442  r = *it;
443  r.ts = getMillisecondsTime();
444  return true;
445  }
446  }
447  return false;
448 }

521  return;
522  }

535  }
536 
537  if (group->member)
538  {
539  groups_lock.unlock();
540  ROS_ERROR("GOT DISCONNECT MSG FROM DOWNLINK: GROUP[%s] DOWNLINK[%s]", f->mc_group_.c_str(), getHostnameFromMac(f->eh_h_.eh_source).c_str());
541  groups_lock.lock();
